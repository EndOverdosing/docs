Try any endpoint live without writing any code. Enter a TMDB ID below and browse sources, download links, subtitles, and ffmpeg commands returned by the API.

<Tip>
  Not sure what TMDB ID to use? Try `550` (Fight Club) or `27205` (Inception) for movies, or `1396` (Breaking Bad) for TV.
</Tip>

<iframe
  id="vyla-playground"
  src="https://vyla-api.pages.dev/api/playground"
  width="100%"
  height="800"
  frameborder="0"
  style={{ borderRadius: "8px", display: "block", border: "none" }}
  onLoad={() => {
    window._vylaIframe = document.getElementById('vyla-playground');

    function sendTheme() {
      const html = document.documentElement;
      const isDark =
        html.classList.contains('dark') ||
        html.getAttribute('data-theme') === 'dark' ||
        html.getAttribute('data-color-scheme') === 'dark';
      if (window._vylaIframe?.contentWindow) {
        window._vylaIframe.contentWindow.postMessage({ type: 'theme', dark: isDark }, '*');
      }
    }

    sendTheme();

    if (window._vylaObserver) window._vylaObserver.disconnect();
    window._vylaObserver = new MutationObserver(sendTheme);
    window._vylaObserver.observe(document.documentElement, {
      attributes: true,
      attributeFilter: ['class', 'data-theme', 'data-color-scheme']
    });
  }}
/>
